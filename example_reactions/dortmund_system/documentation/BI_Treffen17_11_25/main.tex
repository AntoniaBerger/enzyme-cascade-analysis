\begin{minipage}{0.50\textwidth}
\centering
     \begin{block}{Experimentaler Aufblauf}
       \begin{equation*}
           \min_{\theta} \sum_i \left(\frac{f(c_i,\theta) - r_i}{ \sigma_i}\right)^2
       \end{equation*}
       \begin{itemize}
        \item $c_i$: Konzentrationsmessung zum Zeitpunkt $i$
        \item $f(c_i,\theta) = \frac{c_i \cdot Vmax}{c_i + K_m}$: Modelfunktion
        \item $\theta = (Vmax, K_m)$: Parameter
        \item $\sigma_i$: Standardabweichung der Messung
       \end{itemize}
   \end{block}
\end{minipage}
